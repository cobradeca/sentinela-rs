<script>
    function darkModeToggle() {
        return {
            currentTheme: 'light',
            init() {
                // Aguarda o DarkMode estar disponível
                const checkDarkMode = setInterval(() => {
                    if (window.DarkMode) {
                        clearInterval(checkDarkMode);
                        const html = document.documentElement;
                        this.currentTheme = html.classList.contains('dark') ? 'dark' : 'light';
                    }
                }, 10);
            },
            toggle() {
                if (window.DarkMode) {
                    window.DarkMode.toggle();
                    this.currentTheme = window.DarkMode.getCurrentTheme();
                }
            },
            onThemeChanged(event) {
                if (event.detail) {
                    this.currentTheme = event.detail.theme;
                }
            }
        }
    }
</script>

<script>
    function toggleNotifications() {
        const panel = document.getElementById('notification-panel');
        panel.classList.toggle('hidden');
    }

    function toggleAccordion(header) {
        const contentId = header.nextElementSibling.id; // Pega o ID do conteúdo seguinte
        const content = document.getElementById(contentId);
        const chevron = header.querySelector('i'); // Ícone do chevron

        content.classList.toggle('hidden');

        // Rotaciona o chevron
        if (content.classList.contains('hidden')) {
            chevron.style.transform = 'rotate(0deg)';
        } else {
            chevron.style.transform = 'rotate(180deg)';
        }
    }

    // Fecha ao clicar fora
    document.addEventListener('click', function(e) {
        const toggle = document.getElementById('notification-toggle');
        const panel = document.getElementById('notification-panel');
        if (!toggle.contains(e.target) && !panel.contains(e.target)) {
            panel.classList.add('hidden');
        }
    });
</script>

<script>
    function closeDataTableModal() {
        const modal = document.getElementById('dataTableModal');
        document.body.focus();
        modal.classList.add('hidden');
        modal.setAttribute('aria-hidden', 'true');
    }

    function openModal() {
        const modal = document.getElementById('imageModal');
        modal.classList.remove('hidden');
        modal.setAttribute('aria-hidden', 'false');
        // Foca no modal ou em um elemento interno ao abrir
        modal.querySelector('.modal-footer button').focus();
    }

    function closeModal() {
        const modal = document.getElementById('imageModal');
        // Move o foco para o corpo ou outro elemento antes de esconder
        document.body.focus();
        modal.classList.add('hidden');
        modal.setAttribute('aria-hidden', 'true');
    }

    // Fechar ao clicar fora ou pressionar Esc
    document.addEventListener('DOMContentLoaded', () => {
        const modal = document.getElementById('imageModal');
        const dataTableModal = document.getElementById('dataTableModal');

        modal.addEventListener('click', (e) => {
            if (e.target === modal) {
                closeModal();
            }
        });

        dataTableModal.addEventListener('click', (e) => {
            if (e.target === dataTableModal) {
                closeDataTableModal();
            }
        });

        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') {
                closeModal();
                closeDataTableModal();
            }
        });
    });
